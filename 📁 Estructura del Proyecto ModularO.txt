const BASE_URL = 'http://localhost:3000';

/**
 * Función auxiliar para estandarizar las peticiones HTTP (GET, POST, PUT, DELETE)
 */
async function httpRequest(endpoint, options = {}) {
  try {
    const response = await fetch(`${BASE_URL}${endpoint}`, {
      headers: { 'Content-Type': 'application/json', ...options.headers },
      ...options
    });

    if (!response.ok) throw new Error(`Error en la petición: ${response.statusText}`);
    return await response.json();
  } catch (error) {
    console.error(`Error API (${endpoint}):`, error);
    throw error;
  }
}

// Métodos CRUD exportables para el consumo del sistema
export const apiClient = {
  get: (endpoint) => httpRequest(endpoint, { method: 'GET' }),
  post: (endpoint, data) => httpRequest(endpoint, { method: 'POST', body: JSON.stringify(data) }),
  put: (endpoint, data) => httpRequest(endpoint, { method: 'PUT', body: JSON.stringify(data) }),
  delete: (endpoint) => httpRequest(endpoint, { method: 'DELETE' })
};

export function showToast(message, type = 'success') {
  const container = document.getElementById('toast-container') || createToastContainer();
  
  const toast = document.createElement('div');
  toast.className = `p-4 rounded shadow-lg text-white font-medium transition-all duration-300 transform translate-y-2 max-w-sm ${
    type === 'success' ? 'bg-green-600' : 'bg-red-600'
  }`;
  toast.textContent = message;

  container.appendChild(toast);
  setTimeout(() => toast.remove(), 3000);
}

function createToastContainer() {
  const container = document.createElement('div');
  container.id = 'toast-container';
  container.className = 'fixed bottom-5 right-5 z-50 flex flex-col gap-2';
  document.body.appendChild(container);
  return container;
}

/**
 * Función helper para renderizar dinámicamente botones de acción basados en las Reglas de Negocio [cite: 115]
 */
function renderRowActions(user, b) {
  let actions = '';

  if (user.role === 'admin') {
    // El Administrador tiene control absoluto de estados y edición [cite: 119, 128]
    if (b.status === 'Pendiente') {
      actions += `
        <button data-action="approve" data-id="${b.id}" class="bg-blue-500 text-white px-2 py-1 text-xs rounded hover:bg-blue-600">Aprobar</button>
        <button data-action="reject" data-id="${b.id}" class="bg-yellow-500 text-white px-2 py-1 text-xs rounded hover:bg-yellow-600">Rechazar</button>
      `;
    }
    actions += `
      <button data-action="edit" data-id="${b.id}" class="bg-gray-500 text-white px-2 py-1 text-xs rounded hover:bg-gray-600">Editar</button>
      <button data-action="delete" data-id="${b.id}" class="bg-red-500 text-white px-2 py-1 text-xs rounded hover:bg-red-600">Eliminar</button>
    `;
  } else {
    // Reglas para Usuario Estándar [cite: 132]
    if (b.status === 'Pendiente') {
      actions += `
        <button data-action="edit" data-id="${b.id}" class="bg-gray-500 text-white px-2 py-1 text-xs rounded hover:bg-gray-600">Modificar</button>
        <button data-action="cancel" data-id="${b.id}" class="bg-red-400 text-white px-2 py-1 text-xs rounded hover:bg-red-500">Cancelar</button>
      `;
    } else if (b.status === 'Aprobada') {
      // Las reservas aprobadas solo se pueden cancelar [cite: 118]
      actions += `<button data-action="cancel" data-id="${b.id}" class="bg-red-400 text-white px-2 py-1 text-xs rounded hover:bg-red-500">Cancelar</button>`;
    }
  }
  return actions || '<span class="text-xs text-gray-400">Sin acciones</span>';
}

export async function handleRouting() {
  const hash = window.location.hash || '#/login';
  const route = routes[hash];

  // 1. Manejo de Rutas no encontradas (404) -> Redirección por defecto
  if (!route) {
    window.location.hash = '#/login';
    return;
  }

  // Capturamos el usuario activo desde LocalStorage [cite: 64, 179]
  const currentUser = JSON.parse(localStorage.getItem('currentUser'));

  // 2. Guard de Autenticación: Bloquear si es privada y no está logueado [cite: 156, 157]
  if (route.private && !currentUser) {
    showToast('Acceso denegado. Por favor inicia sesión.', 'error'); [cite: 165]
    window.location.hash = '#/login';
    return;
  }

  // 3. Guard de Autorización por Roles: Evitar intrusos en módulos administrativos [cite: 13, 158]
  if (route.roleRequired && currentUser.role !== route.roleRequired) {
    showToast('No tienes permisos para acceder a este módulo', 'error'); [cite: 165]
    window.location.hash = '#/dashboard'; // Redirección de seguridad [cite: 163]
    return;
  }

  // --- RENDERIZADO DINÁMICO DE COMPONENTES --- 
  const appContainer = document.getElementById('app');
  
  // Inyectamos el Navbar superior adaptado y luego la vista correspondiente [cite: 192]
  appContainer.innerHTML = `
    ${renderNavbar()}
    <main class="min-h-screen bg-gray-50">${await route.view.render()}</main>
  `;

  // Inicializamos eventos interactivos del DOM de los módulos activos [cite: 204, 205]
  initNavbarEvents();
  route.view.init();
}
